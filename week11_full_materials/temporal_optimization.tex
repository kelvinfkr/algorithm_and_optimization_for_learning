\section{泛函优化：从优化点到优化函数}
%========================================

\begin{blackboard}[什么是泛函？]

\textbf{回顾}：
\begin{itemize}
    \item \textbf{函数} $f: \mathbb{R}^n \to \mathbb{R}$，输入向量，输出数
    \item 例：$f(x_1, x_2) = x_1^2 + x_2^2$
\end{itemize}

\textbf{新概念}：
\begin{itemize}
    \item \textbf{泛函} $J: \mathcal{F} \to \mathbb{R}$，输入函数，输出数
    \item $\mathcal{F}$ 是某个函数空间（如所有连续函数）
\end{itemize}

\textbf{例子}：
\[
J[y] = \int_a^b F(y(x), y'(x), x) \, dx
\]

输入一个函数 $y(x)$，输出一个数（积分值）

\end{blackboard}

\begin{blackboard}[泛函的具体例子]

\textbf{例1：曲线长度}

从 $(0, 0)$ 到 $(1, 1)$ 的曲线 $y(x)$，长度是多少？

\[
L[y] = \int_0^1 \sqrt{1 + (y'(x))^2} \, dx
\]

这里 $F(y, y', x) = \sqrt{1 + (y')^2}$

\textbf{具体计算}：如果 $y(x) = x$（直线）

$y'(x) = 1$，$L = \int_0^1 \sqrt{1 + 1} \, dx = \sqrt{2} \approx 1.414$

如果 $y(x) = x^2$（抛物线）

$y'(x) = 2x$，$L = \int_0^1 \sqrt{1 + 4x^2} \, dx \approx 1.479$

\textbf{直线更短}！（符合直觉）

\end{blackboard}

\begin{blackboard}[泛函的具体例子（续）]

\textbf{例2：最速降线（Brachistochrone）}

小球从 $(0, 0)$ 滑到 $(1, -1)$，哪条路径最快？

\begin{center}
\begin{tikzpicture}[scale=2]
    \draw[->] (0, 0.2) -- (0, -1.3) node[below] {$y$};
    \draw[->] (-0.2, 0) -- (1.3, 0) node[right] {$x$};
    \fill (0, 0) circle (1pt) node[above left] {$(0,0)$};
    \fill (1, -1) circle (1pt) node[below right] {$(1,-1)$};
    \draw[thick, blue] (0, 0) -- (1, -1) node[midway, above right] {直线};
    \draw[thick, red, domain=0:1, samples=50] plot (\x, {-\x*\x}) node[right] {抛物线};
    \draw[thick, green!50!black, domain=0:1, samples=50] plot (\x, {-0.5*(1-cos(180*\x))}) node[right] {摆线};
\end{tikzpicture}
\end{center}

时间泛函：$T[y] = \int_0^1 \frac{\sqrt{1 + (y')^2}}{\sqrt{-2gy}} \, dx$

\textbf{答案}：摆线（Cycloid）！不是直线也不是抛物线

\end{blackboard}

\begin{blackboard}[泛函的具体例子（续）]

\textbf{例3：信息熵}

给定概率分布 $p(x)$，熵是多少？

\[
H[p] = -\int p(x) \log p(x) \, dx
\]

输入一个概率密度函数 $p(x)$，输出一个数（熵值）

\textbf{例4：KL散度}

两个分布 $p, q$ 的``距离''：

\[
D_{\text{KL}}[p \| q] = \int p(x) \log \frac{p(x)}{q(x)} \, dx
\]

\textbf{这些都是泛函！}

\end{blackboard}

\begin{blackboard}[泛函优化的目标]

\textbf{问题}：在所有满足某些条件的函数中，找使泛函取极值的函数

\textbf{普通优化}：
\[
\min_{x \in \mathbb{R}^n} f(x) \quad \Rightarrow \quad \nabla f(x^*) = 0
\]

\textbf{泛函优化}：
\[
\min_{y(\cdot) \in \mathcal{F}} J[y] \quad \Rightarrow \quad \text{??}
\]

\textbf{需要回答}：

``泛函的梯度''是什么？

``泛函的极值条件''是什么？

\textbf{答案}：Euler-Lagrange 方程！

\end{blackboard}

%========================================
\section{Euler-Lagrange方程：泛函的极值条件}\label{sec:euler-lagrange}
%========================================

\begin{blackboard}[策略：泛函优化作为约束优化]

E-L 方程有两条推导路线。我们\textbf{先走}有限维约束优化那条
（拉格朗日乘子法），\textbf{再}用经典变分法 $\delta J = 0$ 验证。

\textbf{为什么这么排}？
\begin{itemize}
    \item 离散化把问题降回有限维——大一多元微积分就够用
    \item 乘子 $\lambda_k$ 不是辅助记号——连续极限下\textbf{它就是物理动量} $p = \partial L/\partial \dot q$
    \item 这套机器后面 HJB 节直接复用：$\lambda \to $ costate
\end{itemize}

\textbf{目标}：泛函
\[
J[q] = \int_{t_0}^{t_1} L(q, \dot q, t)\, dt
\]
取极值的必要条件。

\end{blackboard}

\begin{blackboard}[Step 1：时间离散化，位置与速度独立成变量]

\textbf{时间剖分}：$t_k = t_0 + k\,\Delta t$，$k = 0, 1, \ldots, N$，$\Delta t = (t_1-t_0)/N$。

\textbf{变量}：位置 $q_k \approx q(t_k)$，速度 $v_k \approx \dot q(t_k)$。

\textbf{关键}：把 $q_k$ 和 $v_k$ 当作\textbf{独立变量}处理。它们之间的关系
\[
v_k = \frac{q_{k+1} - q_k}{\Delta t}
\]
现在变成\textbf{显式约束}，而不是定义。这一招就是后面 $\lambda_k$ 出场的舞台。

\textbf{离散作用量}：
\[
J(\{q_k\}, \{v_k\}) = \sum_{k=0}^{N-1} L(q_k, v_k, t_k)\,\Delta t
\]

\textbf{运动学约束}（每个时间间隔一条）：
\[
g_k := q_{k+1} - q_k - \Delta t\,v_k = 0,\qquad k = 0, 1, \ldots, N-1
\]
共 $N$ 条等式约束。边界 $q_0, q_N$ 由初/末位置固定。

\end{blackboard}

\begin{blackboard}[Step 2：拉格朗日乘子绑定运动学约束]

每条约束 $g_k = 0$ 引入一个乘子 $\lambda_k$，构造\textbf{有限维拉格朗日函数}：
\[
\mathcal{L}(\{q_k\}, \{v_k\}, \{\lambda_k\})
= \sum_{k=0}^{N-1} L(q_k, v_k, t_k)\,\Delta t
+ \sum_{k=0}^{N-1}\lambda_k\bigl(q_{k+1} - q_k - \Delta t\,v_k\bigr)
\]
\textbf{这是一个标准的大一级别多元函数}——变量数 $3N+1$（$N+1$ 个 $q_k$，$N$ 个 $v_k$，$N$ 个 $\lambda_k$）。
做的事情和"用拉格朗日乘子求带约束极值"一模一样。

接下来只有一件事：\textbf{对每个变量求偏导，设为零}。

\end{blackboard}

\begin{blackboard}[Step 3：KKT 条件——三组方程]

\textbf{对 $v_k$ 求偏导}（$k = 0, \ldots, N-1$）：
\[
\frac{\partial \mathcal{L}}{\partial v_k}
= \frac{\partial L}{\partial v_k}\bigg|_k \Delta t \;-\; \lambda_k\,\Delta t = 0
\quad\Longrightarrow\quad
\boxed{\,\lambda_k = \frac{\partial L}{\partial v_k}\bigg|_k\,}
\]
\textbf{这就是共轭动量}。$\lambda$ 在这一步从"约束乘子"获得了物理身份。

\textbf{对内部点 $q_k$ 求偏导}（$1 \le k \le N-1$，$q_k$ 同时出现在第 $k$ 项 $L$ 和约束 $g_{k-1}, g_k$ 里）：
\[
\frac{\partial \mathcal{L}}{\partial q_k}
= \frac{\partial L}{\partial q_k}\bigg|_k\,\Delta t \;-\; \lambda_k \;+\; \lambda_{k-1} = 0
\]
整理：
\[
\boxed{\,\frac{\lambda_k - \lambda_{k-1}}{\Delta t} = \frac{\partial L}{\partial q_k}\bigg|_k\,}
\]
\textbf{乘子的离散时间导数等于 $L$ 对位置的偏导}——这就是即将浮现的 E-L 方程。

\textbf{对 $\lambda_k$ 求偏导}：
\[
\frac{\partial \mathcal{L}}{\partial \lambda_k} = q_{k+1} - q_k - \Delta t\,v_k = 0
\]
恢复运动学约束（这一组等式只是把 $v_k$ 和 $q_k$ 关联起来）。

\end{blackboard}

\begin{blackboard}[Step 4：连续极限 $\to$ Euler-Lagrange 方程]

$\Delta t \to 0$ 时：
\[
\lambda_k \to p(t),\qquad v_k \to \dot q(t),\qquad \frac{\lambda_k-\lambda_{k-1}}{\Delta t} \to \dot p(t)
\]
KKT 三式分别变成：
\begin{align*}
\text{(I)}\quad & p = \frac{\partial L}{\partial \dot q} &&\text{（共轭动量定义）}\\
\text{(II)}\quad & \dot p = \frac{\partial L}{\partial q} &&\text{（动量演化）}\\
\text{(III)}\quad & \dot q = v &&\text{（运动学约束）}
\end{align*}
把 (I) 代入 (II) 消去 $p$：
\[
\frac{d}{dt}\frac{\partial L}{\partial \dot q} = \frac{\partial L}{\partial q}
\]

即
\[
\boxed{\;\frac{d}{dt}\frac{\partial L}{\partial \dot q_i} - \frac{\partial L}{\partial q_i} = 0\;}
\]
这就是 \textbf{Euler-Lagrange 方程}。
\textbf{物理读法}：\textbf{动量随时间的变化率等于 $L$ 对位置的偏导}——\textbf{广义牛顿第二定律}。

\end{blackboard}

\begin{keypoint}[本质洞察：E-L 方程是拉格朗日乘子的演化方程]

我们不是"凑出"了 E-L 方程，而是把约束优化的 KKT 条件取了连续极限。
其中乘子 $\lambda_k \to p(t)$ 是核心——它在五种语言里有五个名字：
\begin{center}
\begin{tabular}{c|c}
\toprule
语言 & $\lambda$ 的名字 \\
\midrule
约束优化 & 拉格朗日乘子 \\
经典力学 & 共轭动量 $p = \partial L / \partial \dot q$ \\
变分法 & 作用量梯度 $\partial S / \partial q$ \\
最优控制 & costate / 伴随变量（HJB / PMP 节） \\
\bottomrule
\end{tabular}
\end{center}
\textbf{这门课接下来所有的"$\lambda$"都是这一个 $\lambda$}。

\end{keypoint}

\begin{blackboard}[另一种推导：经典变分法（$\delta J = 0$）]

完整起见，给出经典推导。这条路不离散化，直接在函数空间求极值。

\textbf{变分概念}：普通微分扰动 $x \to x + \epsilon$；
泛函变分则扰动整个函数 $q(t) \to q(t) + \epsilon\,\eta(t)$，端点固定 $\eta(t_0) = \eta(t_1) = 0$。

\textbf{一阶变分}：
\[
\delta J = \epsilon \int_{t_0}^{t_1}\left( \frac{\partial L}{\partial q}\eta + \frac{\partial L}{\partial \dot q}\dot\eta \right) dt
\]

\textbf{第二项分部积分}（端点项因 $\eta(t_0) = \eta(t_1) = 0$ 而消失）：
\[
\int \frac{\partial L}{\partial \dot q}\dot\eta\, dt = -\int \frac{d}{dt}\frac{\partial L}{\partial \dot q}\,\eta\, dt
\]

\textbf{合并}：
\[
\delta J = \epsilon \int_{t_0}^{t_1} \eta\,\left(\frac{\partial L}{\partial q} - \frac{d}{dt}\frac{\partial L}{\partial \dot q}\right) dt
\]

对所有满足边界条件的 $\eta$，$\delta J = 0$ $\Longrightarrow$ 括号内必为零：
\[
\frac{d}{dt}\frac{\partial L}{\partial \dot q} - \frac{\partial L}{\partial q} = 0\quad\checkmark
\]
和拉格朗日乘子法\textbf{殊途同归}。

\end{blackboard}

\begin{blackboard}[两种推导方法对比]
\begin{center}
\begin{tabular}{c|c|c}
\toprule
 & \textbf{拉格朗日乘子法} & \textbf{经典变分法} \\
\midrule
起点 & 离散化 + 显式运动学约束 & 函数空间的扰动 \\
工具 & KKT 条件 & 分部积分 \\
$\lambda$ 角色 & \textbf{共轭动量 $p$}（核心物理量） & 不显式出现 \\
连续极限 & $\Delta t \to 0$ & 无（直接得到） \\
后续衔接 & 直通 HJB / costate / 伴随 & 适合纯理论分析 \\
\bottomrule
\end{tabular}
\end{center}

\textbf{选哪条}？两条都得到同一个 E-L 方程，但拉格朗日乘子法
\textbf{保留了 $\lambda = p$ 这条物理线索}——后面 Hamilton 力学、最优控制（HJB / PMP）都用得到。经典变分法更简洁，
但 $\lambda$ 隐藏在分部积分里没有显式出现。

\end{blackboard}


\begin{blackboard}[例1：最短路径]

\textbf{问题}：从 $(0, y_0)$ 到 $(1, y_1)$，哪条曲线最短？

\[
L[y] = \int_0^1 \sqrt{1 + (y')^2} \, dx
\]

这里 $F = \sqrt{1 + (y')^2}$

\textbf{计算偏导}：

$\frac{\partial F}{\partial y} = 0$（$F$ 不显含 $y$）

$\frac{\partial F}{\partial y'} = \frac{y'}{\sqrt{1 + (y')^2}}$

\textbf{Euler-Lagrange方程}：

\[
0 - \frac{d}{dx}\left( \frac{y'}{\sqrt{1 + (y')^2}} \right) = 0
\]

$\Rightarrow \frac{y'}{\sqrt{1 + (y')^2}} = C$（常数）

$\Rightarrow y' = $ 常数 $\Rightarrow y = ax + b$（\textbf{直线}！）

\end{blackboard}

\begin{blackboard}[例1数值验证]

\textbf{边界条件}：$(0, 0) \to (1, 1)$

\textbf{E-L解}：$y = x$（直线）

$L = \int_0^1 \sqrt{1 + 1} \, dx = \sqrt{2} \approx \mathbf{1.414}$

\textbf{尝试其他曲线}：

$y = x^2$：$y' = 2x$

$L = \int_0^1 \sqrt{1 + 4x^2} \, dx = \frac{1}{4}\left[ 2x\sqrt{1+4x^2} + \sinh^{-1}(2x) \right]_0^1 \approx \mathbf{1.479}$

$y = \sin(\frac{\pi x}{2})$：$y' = \frac{\pi}{2}\cos(\frac{\pi x}{2})$

$L = \int_0^1 \sqrt{1 + \frac{\pi^2}{4}\cos^2(\frac{\pi x}{2})} \, dx \approx \mathbf{1.534}$

\textbf{结论}：直线确实最短！

\end{blackboard}

\begin{blackboard}[例2：最速降线]

\textbf{问题}：小球从原点滑到 $(x_1, y_1)$，重力场中，哪条路最快？

速度 $v = \sqrt{2gy}$（能量守恒），时间：
\[
T[y] = \int_0^{x_1} \frac{\sqrt{1 + (y')^2}}{\sqrt{2gy}} \, dx
\]

这里 $F = \frac{\sqrt{1 + (y')^2}}{\sqrt{2gy}}$

\textbf{E-L方程的解}：摆线（Cycloid）参数形式
\[
x = r(\theta - \sin\theta), \quad y = r(1 - \cos\theta)
\]

\textbf{历史}：1696年，Johann Bernoulli提出，Newton、Leibniz、L'Hôpital、Jakob Bernoulli都给出了解答

\end{blackboard}

\begin{blackboard}[例2数值验证]

\textbf{边界}：$(0, 0) \to (1, 1)$，$g = 10$

\textbf{直线} $y = x$：
\[
T_{\text{直线}} = \int_0^1 \frac{\sqrt{2}}{\sqrt{20x}} \, dx = \frac{\sqrt{2}}{\sqrt{20}} \cdot 2\sqrt{x}\Big|_0^1 = \frac{2\sqrt{2}}{\sqrt{20}} \approx \mathbf{0.632} \text{ s}
\]

\textbf{摆线}（数值积分）：
\[
T_{\text{摆线}} \approx \mathbf{0.584} \text{ s}
\]

\textbf{摆线比直线快约 8\%！}

\textbf{直觉}：摆线开始时更陡，小球先获得更大速度

\end{blackboard}

\begin{blackboard}[Beltrami恒等式：特殊情况]

当 $F$ 不显含 $x$ 时（$\frac{\partial F}{\partial x} = 0$），有更简单的形式：

\[
\boxed{F - y' \frac{\partial F}{\partial y'} = C \quad \text{（常数）}}
\]

\textbf{证明}（选读）：
\[
\frac{d}{dx}\left( F - y' \frac{\partial F}{\partial y'} \right) = \frac{\partial F}{\partial y}y' + \frac{\partial F}{\partial y'}y'' - y''\frac{\partial F}{\partial y'} - y'\frac{d}{dx}\frac{\partial F}{\partial y'}
\]
\[
= y' \left( \frac{\partial F}{\partial y} - \frac{d}{dx}\frac{\partial F}{\partial y'} \right) = 0
\]

\textbf{应用}：最速降线、测地线等问题可以用这个简化

\end{blackboard}

\begin{blackboard}[例3：悬链线（Catenary）]

\textbf{问题}：两端固定的柔软链条，在重力下的形状是什么？

\begin{center}
\begin{tikzpicture}[scale=1.2]
    \draw[thick] (-2, 1.5) -- (-2, 0);
    \draw[thick] (2, 1.5) -- (2, 0);

\textbf{物理}：链条在势能最低的位置平衡

\textbf{泛函}：势能 = $\int \rho g y \, ds = \int \rho g y \sqrt{1 + (y')^2} \, dx$

这里 $F = y \sqrt{1 + (y')^2}$

\end{blackboard}

\begin{blackboard}[悬链线的求解]

\textbf{用Beltrami恒等式}（$F$ 不显含 $x$）：

$F - y' \frac{\partial F}{\partial y'} = C$

\textbf{计算}：
\[
\frac{\partial F}{\partial y'} = \frac{y \cdot y'}{\sqrt{1 + (y')^2}}
\]

\[
F - y' \frac{\partial F}{\partial y'} = y\sqrt{1 + (y')^2} - \frac{y(y')^2}{\sqrt{1 + (y')^2}} = \frac{y}{\sqrt{1 + (y')^2}} = C
\]

\textbf{解出}：$y = C \sqrt{1 + (y')^2}$

令 $y' = \sinh(t)$，则 $y = C \cosh(t)$

\textbf{悬链线方程}：$y = a \cosh\left(\frac{x}{a}\right)$

\end{blackboard}

\begin{blackboard}[悬链线数值验证]

\textbf{边界}：$(-1, 1.5) \to (1, 1.5)$

\textbf{悬链线}：$y = a \cosh(x/a)$，其中 $a \cosh(1/a) = 1.5$

数值求解：$a \approx 0.83$

$y(x) = 0.83 \cosh(x/0.83)$

\textbf{势能}（取 $\rho g = 1$）：
\[
E_{\text{悬链线}} = \int_{-1}^{1} y \sqrt{1 + (y')^2} \, dx \approx \mathbf{3.52}
\]

\textbf{对比抛物线}：$y = 0.5x^2 + 1$（同样经过两端点）
\[
E_{\text{抛物线}} = \int_{-1}^{1} (0.5x^2 + 1) \sqrt{1 + x^2} \, dx \approx \mathbf{3.58}
\]

\textbf{悬链线势能更低}！（3.52 < 3.58）

\end{blackboard}

\begin{blackboard}[例4：等周问题（Isoperimetric）]

\textbf{问题}：给定周长 $L$，围成最大面积的曲线是什么？

\textbf{约束}：$\oint ds = L$（周长固定）

\textbf{目标}：$\max A = \frac{1}{2}\oint (x \, dy - y \, dx)$

\textbf{这是带约束的泛函优化！}

\textbf{用拉格朗日乘子}：
\[
\mathcal{L} = A - \lambda (周长 - L)
\]

\textbf{结论}：曲率恒定 $\Rightarrow$ \textbf{圆}！

\textbf{数值验证}：周长 $L = 2\pi$

圆：半径 $r = 1$，面积 $= \pi \approx \mathbf{3.14}$

正方形：边长 $= \pi/2$，面积 $= (\pi/2)^2 \approx \mathbf{2.47}$

正六边形：面积 $\approx \mathbf{2.72}$

\textbf{圆确实最大}！

\end{blackboard}

\begin{blackboard}[多变量推广]

\textbf{多个函数}：$J[y_1, \ldots, y_n] = \int F(y_1, \ldots, y_n, y_1', \ldots, y_n', x) \, dx$

\textbf{E-L方程}（对每个 $y_i$）：
\[
\frac{\partial F}{\partial y_i} - \frac{d}{dx}\frac{\partial F}{\partial y_i'} = 0, \quad i = 1, \ldots, n
\]

\textbf{E-L方程}：
\[
\frac{\partial F}{\partial y} - \frac{d}{dx}\frac{\partial F}{\partial y'} + \frac{d^2}{dx^2}\frac{\partial F}{\partial y''} = 0
\]

\end{blackboard}

\begin{blackboard}[Hamilton原理：动力学源于变分]

E-L 之前的例子都是\textbf{几何/静力}泛函（路径长度、势能极小化）。
但 E-L 真正诞生于\textbf{经典力学}——这是 Lagrange 1788 年的工作。

\textbf{Hamilton 原理（1834）}：物理系统的真实轨迹 $q(t)$ 使\textbf{作用量}泛函取极值
\[
S[q] = \int_{t_0}^{t_1} L(q, \dot q, t)\, dt, \qquad L = T - V
\]
其中 $T$ 是动能、$V$ 是势能——都是\textbf{标量}。运动方程就是 E-L：
\[
\frac{d}{dt}\frac{\partial L}{\partial \dot q_i} - \frac{\partial L}{\partial q_i} = 0
\]

\textbf{奇迹}：你只要写出两个\textbf{标量}（$T$ 和 $V$），求几次偏导，运动方程就自动浮现——
不画受力图、不分解矢量、不算约束反作用力。

\end{blackboard}

\begin{blackboard}[例5：单摆——牛顿 vs 拉格朗日（热身）]

长度 $\ell$、质量 $m$ 的单摆，求 $\theta(t)$。

\textbf{牛顿法}：画受力图。重力 $mg$ 竖直、张力 $T_\text{绳}$ 沿绳指向支点。
\begin{itemize}
    \item 切向分解：$m\ell\ddot\theta = -mg\sin\theta$
    \item 径向：$T_\text{绳} - mg\cos\theta = m\ell\dot\theta^2$（要算，但和运动方程无关）
\end{itemize}
\textbf{必须}选对分解方向，\textbf{必须}手动丢掉张力。

\textbf{拉格朗日法}：广义坐标 $\theta$，位置 $(\ell\sin\theta, -\ell\cos\theta)$，
速度平方 $\dot x^2 + \dot y^2 = \ell^2\dot\theta^2$。直接写：
\[
T = \tfrac{1}{2}m\ell^2\dot\theta^2,\qquad
V = -mg\ell\cos\theta,\qquad
L = \tfrac{1}{2}m\ell^2\dot\theta^2 + mg\ell\cos\theta
\]
代 E-L：
\[
\frac{d}{dt}(m\ell^2\dot\theta) + mg\ell\sin\theta = 0
\quad\Rightarrow\quad
\ddot\theta = -\tfrac{g}{\ell}\sin\theta\ \checkmark
\]
\textbf{张力压根没出现}——约束 $|r|=\ell$ 被广义坐标 $\theta$ 内化掉了。
单摆里两种方法难度差不多；下一个例子（双摆）会把差距拉开一个量级。

\end{blackboard}

\begin{insight}[衔接 HJB：从 Lagrangian 到 Hamiltonian]

连续极限下 $\lambda \to p$。做 \textbf{Legendre 变换}定义 Hamiltonian：

H(q, p, t) := p^\top \dot q - L(q, \dot q, t)\Big|_{\dot q = \dot q(q, p, t)}

（自治保守系统：$H = T + V$ 就是总能量。）E-L 方程化为对称的一阶系统：
\[
\dot q = \frac{\partial H}{\partial p},\qquad
\dot p = -\frac{\partial H}{\partial q}
\]

后面第 \ref{sec:hjb} 节的 HJB 方程
\[
-\frac{\partial V}{\partial t} = \min_u\left\{\ell(x,u) + \nabla_x V \cdot f(x, u)\right\}

\[
L \;\leftrightarrow\; \ell(x, u),\qquad
p \;\leftrightarrow\; \nabla_x V,\qquad
H \;\leftrightarrow\; \min_u\{\ell + p\cdot f\}.
\]

\textbf{变分法、约束优化、Hamilton 力学、最优控制——同一个故事的四种语言。}
你现在写的 $\lambda_k$，就是这个故事的开头。
\end{insight}

\section{最大熵原理：约束下的分布选择}
%========================================

\begin{blackboard}[熵的回顾]

\textbf{离散熵}：$H(p) = -\sum_i p_i \log p_i$

\textbf{连续熵（微分熵）}：$H[p] = -\int p(x) \log p(x) \, dx$

\textbf{熵的含义}：
\begin{itemize}
    \item 不确定性的度量
    \item 信息量的期望
    \item 分布的``扩散程度''
\end{itemize}

\textbf{熵最大的分布}：最``均匀''、最``无偏''

\end{blackboard}

\begin{blackboard}[最大熵原理]

\textbf{问题}：已知某些约束（如均值、方差），应该选哪个分布？

\textbf{最大熵原理}（Jaynes, 1957）：

选择满足约束的\textbf{熵最大}的分布

\textbf{理由}：
\begin{itemize}
    \item 不引入额外假设
    \item 最``保守''的选择
    \item 信息论上最优
\end{itemize}

\textbf{数学形式}：
\[
\max_{p} H[p] = -\int p(x) \log p(x) \, dx
\]
subject to 约束条件

\end{blackboard}

\begin{blackboard}[例1：无约束最大熵]

\textbf{问题}：在 $[a, b]$ 上，哪个分布熵最大？

\textbf{约束}：$\int_a^b p(x) \, dx = 1$（归一化）

\textbf{Lagrangian}：
\[
\mathcal{L}[p] = -\int p \log p \, dx - \lambda \left( \int p \, dx - 1 \right)
\]

\textbf{变分}：$\frac{\delta \mathcal{L}}{\delta p} = -\log p - 1 - \lambda = 0$

$\Rightarrow p(x) = e^{-1-\lambda} = $ 常数

\textbf{归一化}：$p(x) = \frac{1}{b-a}$（\textbf{均匀分布}！）

\end{blackboard}

\begin{blackboard}[例1数值验证]

\textbf{区间} $[0, 1]$

\textbf{均匀分布}：$p(x) = 1$

$H = -\int_0^1 1 \cdot \log 1 \, dx = 0$（以 $e$ 为底）

\textbf{换成 bits}：$H = \log_2(1) = 0$ bits... 不对！

\textbf{更正}：微分熵可以是负数！对于均匀分布：
\[
H = -\int_0^1 1 \cdot \ln 1 \, dx = 0 \text{ nats}
\]

\textbf{对比三角分布}：$p(x) = 2x$（$x \in [0,1]$）
\[
H = -\int_0^1 2x \ln(2x) \, dx = -\int_0^1 2x(\ln 2 + \ln x) \, dx
\]
\[
= -\ln 2 - \int_0^1 2x \ln x \, dx = -\ln 2 - 2 \cdot (-\frac{1}{4}) = -\ln 2 + 0.5 \approx -0.193
\]

\textbf{均匀分布熵更大}（0 > -0.193）

\end{blackboard}

\begin{blackboard}[例2：给定均值的最大熵]

\textbf{问题}：$x \geq 0$，已知 $\mathbb{E}[x] = \mu$，哪个分布熵最大？

\textbf{约束}：
\begin{itemize}
    \item $\int_0^\infty p(x) \, dx = 1$
    \item $\int_0^\infty x \cdot p(x) \, dx = \mu$
\end{itemize}

\textbf{Lagrangian}：
\[
\mathcal{L} = -\int p \log p \, dx - \lambda_0 \left( \int p \, dx - 1 \right) - \lambda_1 \left( \int xp \, dx - \mu \right)
\]

\textbf{变分}：$-\log p - 1 - \lambda_0 - \lambda_1 x = 0$

$\Rightarrow p(x) = e^{-1-\lambda_0} \cdot e^{-\lambda_1 x} = C e^{-\lambda_1 x}$

\textbf{这是指数分布！}$p(x) = \frac{1}{\mu} e^{-x/\mu}$

\end{blackboard}

\begin{blackboard}[例2数值验证]

\textbf{给定} $\mu = 2$

\textbf{指数分布}：$p(x) = 0.5 e^{-x/2}$

熵：$H = -\int_0^\infty p \ln p \, dx = 1 + \ln \mu = 1 + \ln 2 \approx \mathbf{1.693}$ nats

\textbf{对比 Gamma 分布}：$p(x) = \frac{x}{\theta^2} e^{-x/\theta}$，均值 $= 2\theta = 2 \Rightarrow \theta = 1$

$p(x) = x e^{-x}$

熵：$H = 2 - \ln(1) - \psi(2) = 2 - 0 - (1 - \gamma) \approx 2 - 0.423 = \mathbf{1.577}$ nats

\textbf{指数分布熵更大}（1.693 > 1.577）

\end{blackboard}

\begin{blackboard}[例3：给定均值和方差的最大熵]

\textbf{问题}：$x \in \mathbb{R}$，已知 $\mathbb{E}[x] = \mu$，$\text{Var}(x) = \sigma^2$

\textbf{约束}：
\begin{itemize}
    \item $\int p \, dx = 1$
    \item $\int x p \, dx = \mu$
    \item $\int (x - \mu)^2 p \, dx = \sigma^2$
\end{itemize}

\textbf{变分}：$-\log p - 1 - \lambda_0 - \lambda_1 x - \lambda_2 (x-\mu)^2 = 0$

$\Rightarrow p(x) \propto \exp\left( -\lambda_1 x - \lambda_2 (x-\mu)^2 \right)$

配方得：$p(x) = \frac{1}{\sqrt{2\pi\sigma^2}} \exp\left( -\frac{(x-\mu)^2}{2\sigma^2} \right)$

\textbf{这是高斯分布！}

\end{blackboard}

\begin{blackboard}[最大熵与指数族]

\begin{keypoint}[最大熵分布的一般形式]
约束 $\mathbb{E}[f_k(x)] = \alpha_k$，$k = 1, \ldots, K$ 下的最大熵分布是：
\[
\boxed{p(x) = \frac{1}{Z} \exp\left( -\sum_{k=1}^K \lambda_k f_k(x) \right)}
\]
这正是\textbf{指数族分布}的形式！
\end{keypoint}

\textbf{经典结果}：
\begin{center}
\begin{tabular}{c|c|c}
约束 & 最大熵分布 & 名称 \\
\hline
无 & 均匀 & Uniform \\
$\mathbb{E}[x] = \mu$ ($x \geq 0$) & 指数 & Exponential \\
$\mathbb{E}[x], \text{Var}(x)$ & 高斯 & Gaussian \\
$\mathbb{E}[\log x], \mathbb{E}[x]$ & Gamma & Gamma \\
$p_i \geq 0, \sum p_i = 1$ & 离散均匀 & Categorical
\end{tabular}
\end{center}

\end{blackboard}

\begin{blackboard}[为什么高斯分布如此常见？]

\textbf{最大熵解释}：

如果只知道均值和方差，高斯是``最保守''的选择

\textbf{中心极限定理解释}：

大量独立随机变量之和趋向高斯

\textbf{两者的联系}：

CLT说高斯是``最可能''出现的

MaxEnt说高斯是``最少假设''的

\textbf{都指向同一个分布}！

\end{blackboard}

\section{KL散度与变分推断}
%========================================

\begin{blackboard}[KL散度的定义]

\textbf{KL散度}（Kullback-Leibler Divergence）：
\[
D_{\text{KL}}(p \| q) = \int p(x) \log \frac{p(x)}{q(x)} \, dx = \mathbb{E}_{p}\left[ \log \frac{p(x)}{q(x)} \right]
\]

\textbf{性质}：
\begin{itemize}
    \item $D_{\text{KL}}(p \| q) \geq 0$（非负）
    \item $D_{\text{KL}}(p \| q) = 0$ 当且仅当 $p = q$
    \item \textbf{不对称}：$D_{\text{KL}}(p \| q) \neq D_{\text{KL}}(q \| p)$
\end{itemize}

\textbf{直觉}：用 $q$ 编码来自 $p$ 的数据，平均多花多少 bits

\end{blackboard}

\begin{blackboard}[Huffman 例子（一）：先看最优编码长什么样]

\textbf{设定}：4 个符号 $\{A, B, C, D\}$。

真实分布 $p$ 和误以为的分布 $q$：
\begin{center}
\begin{tabular}{c|cccc}
\toprule
 & $A$ & $B$ & $C$ & $D$ \\
\midrule
$p$ & $\tfrac{1}{2}$ & $\tfrac{1}{4}$ & $\tfrac{1}{8}$ & $\tfrac{1}{8}$ \\
$q$ & $\tfrac{1}{8}$ & $\tfrac{1}{8}$ & $\tfrac{1}{4}$ & $\tfrac{1}{2}$ \\
\bottomrule
\end{tabular}
\end{center}
（注意 $q$ 把 $A, B$ 当稀有事件，$D$ 当最常见——和 $p$ 几乎反着来。）

\textbf{根据 $p$ 构造 Huffman 编码}：合并两个最小 $\tfrac{1}{8}\to\tfrac{1}{4}$，再合并两个 $\tfrac{1}{4}\to\tfrac{1}{2}$，再合并两个 $\tfrac{1}{2}$：
\[
\text{码表 (p-optimal)}:\quad A: \mathtt{0},\ \ B: \mathtt{10},\ \ C: \mathtt{110},\ \ D: \mathtt{111}
\]
平均长度（在 $p$ 上）：
\[
\bar\ell_p = \tfrac{1}{2}\!\cdot\!1 + \tfrac{1}{4}\!\cdot\!2 + \tfrac{1}{8}\!\cdot\!3 + \tfrac{1}{8}\!\cdot\!3 = 1.75 \text{ bits}\ =\ H(p)\ \checkmark
\]
（恰好相等：$p$ 是 \textbf{dyadic 分布}，Huffman 严格达到熵下界。）

\end{blackboard}

\begin{blackboard}[Huffman 例子（二）：用错码本要多花多少 bits]

\textbf{根据 $q$ 构造 Huffman 编码}：同样的算法，但合并顺序变了——
两个 $\tfrac{1}{8}$（$A,B$）合并成 $\tfrac{1}{4}$，再和 $\tfrac{1}{4}$（$C$）合并成 $\tfrac{1}{2}$，再和 $\tfrac{1}{2}$（$D$）合并：
\[
\text{码表 (q-optimal)}:\quad D: \mathtt{0},\ \ C: \mathtt{10},\ \ A: \mathtt{110},\ \ B: \mathtt{111}
\]
码长：$A\!:\!3,\ B\!:\!3,\ C\!:\!2,\ D\!:\!1$。

\textbf{用这套（错的）$q$-码本编码 $p$ 的数据}——这正是 KL 描述的场景。
平均比特数（在 $p$ 上）：
\[
\bar\ell_q^{(p)} = \tfrac{1}{2}\!\cdot\!3 + \tfrac{1}{4}\!\cdot\!3 + \tfrac{1}{8}\!\cdot\!2 + \tfrac{1}{8}\!\cdot\!1
= \tfrac{3}{2} + \tfrac{3}{4} + \tfrac{1}{4} + \tfrac{1}{8}
= 2.625 \text{ bits}
\]

\textbf{多花的 bits}：
\[
\bar\ell_q^{(p)} - \bar\ell_p = 2.625 - 1.75 = \boxed{\,0.875\text{ bits / 符号}\,}
\]

\textbf{直接算 $D_{\mathrm{KL}}(p\|q)$ 验证}：
\[
D_{\mathrm{KL}}(p\|q) = \sum_i p_i\log_2\!\frac{p_i}{q_i}
= \tfrac{1}{2}\log_2 4 + \tfrac{1}{4}\log_2 2 + \tfrac{1}{8}\log_2 \tfrac{1}{2} + \tfrac{1}{8}\log_2 \tfrac{1}{4}
\]
\[
= 1 + \tfrac{1}{4} - \tfrac{1}{8} - \tfrac{1}{4} = \boxed{\,0.875\text{ bits}\,}\quad\checkmark
\]
\textbf{两个 0.875 完全一致}——KL 散度的操作含义。

\end{blackboard}

\begin{keypoint}[为什么是 $\log(p/q)$？]

Shannon 编码定理：用分布 $q$ 编码时，最优码长是 $\ell_i = -\log_2 q_i$。
\begin{itemize}
    \item 用 $p$ 的码本编 $p$ 的数据：每符号 $-\log_2 p_i$ bits，平均 $H(p)$
    \item 用 $q$ 的码本编 $p$ 的数据：每符号 $-\log_2 q_i$ bits，平均 $\mathbb{E}_p[-\log_2 q] = H(p, q)$（\textbf{交叉熵}）
\end{itemize}
两者之差就是 KL：
\[
D_{\mathrm{KL}}(p\|q) \;=\; H(p,q) - H(p) \;=\; \mathbb{E}_p\!\left[\log_2\!\tfrac{p}{q}\right]
\]
\textbf{操作翻译}：每 1000 个 $p$-符号，用 $q$-码本要比用 $p$-码本多花
$1000 \times 0.875 = 875$ bits——这就是"知识错位"的客观代价。

\textbf{深度学习挂钩}：分类问题里的交叉熵损失 $-\sum y_i \log \hat p_i$ 就是 $H(p, q)$
减去常数 $H(p)$ 后剩下的 $D_{\mathrm{KL}}(p\|q)$——最小化交叉熵 = 最小化 KL。
\end{keypoint}

\begin{blackboard}[KL散度的数值例子]

\textbf{两个高斯分布}：$p = \mathcal{N}(0, 1)$，$q = \mathcal{N}(\mu, \sigma^2)$

\[
D_{\text{KL}}(p \| q) = \frac{1}{2}\left( \frac{1}{\sigma^2} + \frac{\mu^2}{\sigma^2} - 1 - \log\frac{1}{\sigma^2} \right)
\]

\textbf{数值}：$p = \mathcal{N}(0, 1)$，$q = \mathcal{N}(1, 1)$

$D_{\text{KL}}(p \| q) = \frac{1}{2}(1 + 1 - 1 - 0) = \mathbf{0.5}$ nats

\textbf{反过来}：$D_{\text{KL}}(q \| p) = \frac{1}{2}(1 + 1 - 1 - 0) = \mathbf{0.5}$ nats

（这个特例下恰好相等，因为方差相同）

\textbf{不同方差}：$p = \mathcal{N}(0, 1)$，$q = \mathcal{N}(0, 2)$

$D_{\text{KL}}(p \| q) = \frac{1}{2}(0.5 + 0 - 1 - \ln 0.5) = \frac{1}{2}(-0.5 + 0.693) = \mathbf{0.097}$

$D_{\text{KL}}(q \| p) = \frac{1}{2}(2 + 0 - 1 - \ln 2) = \frac{1}{2}(1 - 0.693) = \mathbf{0.153}$

\textbf{不对称}！

\end{blackboard}

\begin{blackboard}[变分推断的动机]

\textbf{贝叶斯推断问题}：

已知观测 $\mathbf{x}$，求后验 $p(\mathbf{z} | \mathbf{x})$

\[
p(\mathbf{z} | \mathbf{x}) = \frac{p(\mathbf{x} | \mathbf{z}) p(\mathbf{z})}{p(\mathbf{x})} = \frac{p(\mathbf{x} | \mathbf{z}) p(\mathbf{z})}{\int p(\mathbf{x} | \mathbf{z}) p(\mathbf{z}) \, d\mathbf{z}}
\]

\textbf{问题}：分母的积分通常\textbf{无法计算}！

例：$\mathbf{z}$ 是 1000 维，积分是 1000 重积分

\textbf{解决方案}：
\begin{itemize}
    \item MCMC：采样近似，但慢
    \item \textbf{变分推断}：用简单分布 $q(\mathbf{z})$ 近似 $p(\mathbf{z} | \mathbf{x})$
\end{itemize}

\end{blackboard}

\begin{blackboard}[变分推断：核心思想]

\textbf{目标}：找 $q(\mathbf{z})$ 使得 $q(\mathbf{z}) \approx p(\mathbf{z} | \mathbf{x})$

\textbf{度量近似好坏}：KL散度
\[
D_{\text{KL}}(q \| p) = \int q(\mathbf{z}) \log \frac{q(\mathbf{z})}{p(\mathbf{z} | \mathbf{x})} \, d\mathbf{z}
\]

\textbf{展开}：
\begin{align*}
D_{\text{KL}}(q \| p) &= \int q \log q \, d\mathbf{z} - \int q \log p(\mathbf{z} | \mathbf{x}) \, d\mathbf{z} \\
&= \int q \log q \, d\mathbf{z} - \int q \log \frac{p(\mathbf{x}, \mathbf{z})}{p(\mathbf{x})} \, d\mathbf{z} \\
&= \int q \log q \, d\mathbf{z} - \int q \log p(\mathbf{x}, \mathbf{z}) \, d\mathbf{z} + \log p(\mathbf{x})
\end{align*}

\end{blackboard}

\begin{blackboard}[ELBO：Evidence Lower Bound]

\textbf{整理上式}：
\[
\log p(\mathbf{x}) = D_{\text{KL}}(q \| p) + \underbrace{\int q \log \frac{p(\mathbf{x}, \mathbf{z})}{q(\mathbf{z})} \, d\mathbf{z}}_{\text{ELBO}(\mathbf{q})}
\]

因为 $D_{\text{KL}} \geq 0$：
\[
\log p(\mathbf{x}) \geq \text{ELBO}(q)
\]

\begin{keypoint}[ELBO]
\[
\boxed{\text{ELBO}(q) = \mathbb{E}_q[\log p(\mathbf{x}, \mathbf{z})] - \mathbb{E}_q[\log q(\mathbf{z})]}
\]

\textbf{最小化 KL $\Leftrightarrow$ 最大化 ELBO}
\end{keypoint}

\end{blackboard}

\begin{blackboard}[ELBO 的两种形式：同一量的两副面孔]

\textbf{形式 1（能量 $+$ 熵 / 自由能视角）}：
\[
\mathrm{ELBO}(q) = \mathbb{E}_q[\log p(\mathbf{x}, \mathbf{z})] + H[q]
\]

\textbf{形式 2（重构 $-$ 正则 / 自编码器视角）}：
\[
\mathrm{ELBO}(q) = \mathbb{E}_q[\log p(\mathbf{x} | \mathbf{z})] - D_{\mathrm{KL}}(q(\mathbf{z}) \| p(\mathbf{z}))
\]

\textbf{数学上完全等价}——一行就推完。把联合写成条件 $\times$ 先验：
$\log p(\mathbf{x}, \mathbf{z}) = \log p(\mathbf{x}|\mathbf{z}) + \log p(\mathbf{z})$，代入形式 1：
\[
\underbrace{\mathbb{E}_q[\log p(\mathbf{x}|\mathbf{z})]}_{\text{重构项}}
+ \underbrace{\mathbb{E}_q[\log p(\mathbf{z})] - \mathbb{E}_q[\log q(\mathbf{z})]}_{=\,-D_{\mathrm{KL}}(q\|p(\mathbf{z}))}
\]
就是形式 2。\textbf{但强调的东西完全不同}——下面三页详谈。

\end{blackboard}

\begin{blackboard}[形式 1 vs 形式 2：四个维度的对照]

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{p{0.20\linewidth}|p{0.35\linewidth}|p{0.35\linewidth}}
\toprule
 & \textbf{形式 1（能量 $+$ 熵）} & \textbf{形式 2（重构 $-$ 正则）} \\
\midrule
\textbf{物理类比} & Helmholtz 自由能 $F = U - TS$ & 率失真：rate $=$ KL，distortion $= -\mathbb{E}[\log p(\mathbf{x}|\mathbf{z})]$ \\
\midrule
\textbf{需要的输入} & 只需联合 $p(\mathbf{x}, \mathbf{z})$ & 必须显式分解 $p(\mathbf{x}|\mathbf{z}) \cdot p(\mathbf{z})$ \\
\midrule
\textbf{$\mathbf{z}$ 的地位} & 普通隐变量，无先验区分 & "潜变量"，有显式先验 $p(\mathbf{z})$ \\
\midrule
\textbf{易算的那一项} & $H[q]$：指数族 $q$ 有闭式（高斯 $\tfrac{1}{2}\log\det\Sigma + c$） & $D_{\mathrm{KL}}(q\|p(\mathbf{z}))$：高斯-高斯有闭式（VAE 招牌） \\
\midrule
\textbf{难算的那一项} & $\mathbb{E}_q[\log p(\mathbf{x},\mathbf{z})]$：MC 方差大 & $\mathbb{E}_q[\log p(\mathbf{x}|\mathbf{z})]$：需要 reparameterization \\
\bottomrule
\end{tabular}
\end{center}

\textbf{两种形式的"易算项"是同一个东西的两种切法}——都是"关于 $q$ 自己或 $q$ 与先验之间的解析项"。
指数族 $q$ 下这一项必有闭式；涉及生成模型的那一项必须采样。

\end{blackboard}

\begin{blackboard}[形式 1 vs 形式 2：优化时的行为差异]

\textbf{形式 1}：能量低 $+$ $q$ 散开。没有显式的"$q$ 要靠近某个固定分布"压力——$q$ 只被联合似然牵引。

\textbf{形式 2}：重构好 $+$ $q(\mathbf{z}|\mathbf{x})$ 不离先验 $p(\mathbf{z})$ 太远。
这个"靠近先验"压力是 VAE 的\textbf{双刃剑}：
\begin{itemize}
    \item \textbf{后验崩溃}（posterior collapse）：当解码器足够强（如 PixelCNN/Transformer）时，
    KL 项把 $q(\mathbf{z}|\mathbf{x})$ 拉到等于 $p(\mathbf{z})$，$\mathbf{z}$ 不传递任何关于 $\mathbf{x}$ 的信息——
    VAE 退化成无 latent 的纯生成模型
    \item $\beta$\textbf{-VAE}（Higgins 2017）：给 KL 项加权重 $\beta$；$\beta > 1$ 鼓励 disentangled latent，
    $\beta < 1$ 缓解后验崩溃
    \item \textbf{KL annealing}：训练早期把 KL 项权重从 $0$ 缓升到 $1$，让重构先学起来
\end{itemize}

\textbf{为什么形式 1 没有这个病}？形式 1 的熵项 $-\mathbb{E}_q[\log q]$ 鼓励 $q$ 散开，但
\textbf{不指定散向哪里}。形式 2 把它拆成 $-\mathbb{E}_q[\log q] + \mathbb{E}_q[\log p(\mathbf{z})]$，
第二项引入了"瞄准 $p(\mathbf{z})$"的指向性。
\textbf{后验崩溃是形式 2 视角放大出来的工程问题}——不是 ELBO 本身的问题。

\end{blackboard}

\begin{keypoint}[选哪个？]

\renewcommand{\arraystretch}{1.15}
\begin{tabular}{p{0.40\linewidth}|p{0.55\linewidth}}
\toprule
\textbf{场景} & \textbf{用哪个形式} \\
\midrule
理论分析、连接 KL/MaxEnt/统计力学 & 形式 1 \\
平均场 VI（$q = \prod_i q_i$，每个 $H[q_i]$ 独立） & 形式 1 \\
能量基模型（EBM）、Boltzmann 机 & 形式 1 \\
Friston 自由能原理（神经科学） & 形式 1 \\
\midrule
写 VAE 代码 / 实现 encoder-decoder & 形式 2 \\
$\beta$-VAE、disentanglement 研究 & 形式 2 \\
Diffusion model（每步 KL 是相邻 timestep 间） & 形式 2 \\
任何"有 encoder 和 decoder 两个网络"的设定 & 形式 2 \\
\bottomrule
\end{tabular}

\vspace{0.5em}
\textbf{助记}：形式 1 长得像\textbf{物理}（能量减熵 $=$ 自由能），形式 2 长得像\textbf{工程}（信号 $-$ 正则）。
做\textbf{理论}用 1，写\textbf{代码}用 2。

\textbf{这正是 VAE 的损失函数（形式 2）}——下一页给出完整 VAE 损失。

\end{keypoint}

\begin{blackboard}[VAE损失函数的来源]

\textbf{变分自编码器（VAE）}：

编码器：$q_\phi(\mathbf{z} | \mathbf{x})$（用神经网络参数化）

解码器：$p_\theta(\mathbf{x} | \mathbf{z})$

\textbf{损失函数}：
\[
\mathcal{L}(\theta, \phi) = -\text{ELBO} = -\mathbb{E}_{q_\phi}[\log p_\theta(\mathbf{x} | \mathbf{z})] + D_{\text{KL}}(q_\phi(\mathbf{z} | \mathbf{x}) \| p(\mathbf{z}))
\]

\textbf{第一项}：重构损失（让解码器能还原输入）

\textbf{第二项}：KL正则（让编码分布接近先验）

\textbf{变分推断给出了 VAE 的理论基础！}

\end{blackboard}

\begin{blackboard}[数值例子：高斯VAE]

\textbf{设置}：

先验 $p(\mathbf{z}) = \mathcal{N}(0, I)$

编码器输出 $q_\phi(\mathbf{z} | \mathbf{x}) = \mathcal{N}(\mu_\phi(\mathbf{x}), \text{diag}(\sigma^2_\phi(\mathbf{x})))$

\textbf{KL项}（解析形式）：对于 $d$ 维高斯

\[
D_{\text{KL}}(q \| p) = \frac{1}{2} \sum_{j=1}^d \left( \mu_j^2 + \sigma_j^2 - 1 - \log \sigma_j^2 \right)
\]

\textbf{数值}：$d = 2$，$\mu = (1, 0.5)$，$\sigma^2 = (0.5, 0.8)$

$D_{\text{KL}} = \frac{1}{2}[(1 + 0.5 - 1 - \ln 0.5) + (0.25 + 0.8 - 1 - \ln 0.8)]$

$= \frac{1}{2}[(0.5 + 0.693) + (0.05 + 0.223)] = \frac{1}{2}[1.193 + 0.273] = \mathbf{0.733}$

\end{blackboard}

\section{信息瓶颈}
%========================================

\begin{blackboard}[信息瓶颈的动机]

\textbf{问题}：给定输入 $X$ 和目标 $Y$，如何找最好的表示 $Z$？

\textbf{好的表示应该}：
\begin{itemize}
    \item 保留关于 $Y$ 的信息（预测能力）
    \item 压缩 $X$ 的冗余信息（简洁性）
\end{itemize}

\textbf{信息论度量}：
\begin{itemize}
    \item $I(Z; Y)$：$Z$ 关于 $Y$ 的信息（越大越好）
    \item $I(X; Z)$：$Z$ 关于 $X$ 的信息（越小越好，压缩）
\end{itemize}

\end{blackboard}

\begin{blackboard}[信息瓶颈目标]

\textbf{Tishby et al., 1999}：

\[
\boxed{\min_{p(z|x)} I(X; Z) - \beta I(Z; Y)}
\]

\textbf{$\beta > 0$ 控制权衡}：
\begin{itemize}
    \item $\beta$ 大：更看重预测能力
    \item $\beta$ 小：更看重压缩
\end{itemize}

\textbf{等价形式}：
\[
\max_{p(z|x)} I(Z; Y) \quad \text{s.t.} \quad I(X; Z) \leq R
\]

在压缩预算 $R$ 下最大化预测信息

\end{blackboard}

\begin{blackboard}[信息瓶颈与深度学习]

\textbf{深度网络}：$X \to h_1 \to h_2 \to \cdots \to h_L \to Y$

每一层 $h_l$ 可以看作 $X$ 的表示

\textbf{Shwartz-Ziv \& Tishby, 2017}：

训练过程中：
\begin{itemize}
    \item 初期：$I(X; h_l)$ 增加（学习特征）
    \item 后期：$I(X; h_l)$ 减少，$I(h_l; Y)$ 保持（压缩无关信息）
\end{itemize}

\textbf{这叫``信息平面''分析}

（虽然后来有争议，但思想很有影响力）

\end{blackboard}

\begin{blackboard}[信息瓶颈的数值例子]

\textbf{简单设置}：

$X \in \{0, 1, 2, 3\}$（4个值），$Y \in \{0, 1\}$（2个值）

\[
p(x, y) = \begin{cases}
0.2 & (x, y) \in \{(0,0), (1,0), (2,1), (3,1)\} \\
0.1 & (x, y) \in \{(0,1), (1,1), (2,0), (3,0)\}
\end{cases}
\]

\textbf{完美表示}：$Z = X$

$I(X; Z) = H(X) \approx 2$ bits（完全保留）

$I(Z; Y) = I(X; Y) \approx 0.28$ bits

\textbf{压缩表示}：$Z = \mathbf{1}[X \geq 2]$

$I(X; Z) = H(Z) = 1$ bit（压缩一半）

$I(Z; Y) = I(X; Y)$（如果 $Z$ 与 $Y$ 关系不变）

\end{blackboard}

\begin{blackboard}[变分信息瓶颈（VIB）]

\textbf{问题}：$I(X; Z)$ 难以计算

\textbf{VIB}（Alemi et al., 2017）：用变分方法近似

\[
\mathcal{L}_{\text{VIB}} = \mathbb{E}_{p(x,y)}\left[ \mathbb{E}_{q_\phi(z|x)}[-\log p_\theta(y|z)] + \beta D_{\text{KL}}(q_\phi(z|x) \| r(z)) \right]
\]

\textbf{比较 VAE}：
\begin{itemize}
    \item VAE：重构 $x$，正则化到先验
    \item VIB：预测 $y$，正则化到先验
\end{itemize}

\textbf{VIB 用于分类}：更鲁棒、更好的泛化

\end{blackboard}

\section{最优控制与HJB方程}\label{sec:hjb}
%========================================

\begin{blackboard}[最优控制问题]

\textbf{问题设置}：

状态：$\mathbf{x}(t) \in \mathbb{R}^n$

控制：$\mathbf{u}(t) \in \mathbb{R}^m$

动力学：$\dot{\mathbf{x}} = f(\mathbf{x}, \mathbf{u}, t)$

\textbf{目标}：最小化代价
\[
J = \phi(\mathbf{x}(T)) + \int_0^T L(\mathbf{x}, \mathbf{u}, t) \, dt
\]

$\phi$：终端代价，$L$：运行代价

\textbf{找最优控制} $\mathbf{u}^*(t)$ 使 $J$ 最小

\end{blackboard}

\begin{blackboard}[两条独立推导路线]

E-L 方程我们走了两条路（拉格朗日乘子法 $+$ 经典变分法）。
\textbf{HJB 同样有两条路，但角色不同}：

\begin{itemize}
    \item \textbf{Track A（变分法 / PMP）}：延续 E-L 节的乘子思路——
    把动力学约束 $\dot x = f$ 当作"无穷多条等式约束"，用 $\lambda(t)$ 配对。
    一阶必要条件给出 \textbf{Pontryagin 最大值原理}——一组 ODE。
    \item \textbf{Track B（动态规划 / HJB）}：从 Bellman 最优性原理出发，
    定义\textbf{值函数} $V(x, t)$，对其 Taylor 展开取连续极限。
    得到一个 PDE。
\end{itemize}

\textbf{两条路殊途同归}：最后通过身份 $\lambda(t) = \nabla_x V(x(t), t)$ 汇合——
\textbf{PMP 的 costate 就是 HJB 的 value 梯度}。这是前面 E-L 节那个 $\lambda$ 在控制问题里的化身。

\textbf{为什么要走两条}？
\begin{itemize}
    \item PMP 给\textbf{单条最优轨迹}（ODE，无维度灾难）——model-based RL / DDP / iLQR / 神经 ODE
    \item HJB 给\textbf{整个 value 场}（PDE，有维度灾难）——value-based RL / Bellman / TD / Q-learning
\end{itemize}
现实工程里两者都用。本节让你看见它们的统一性。

\end{blackboard}

\begin{blackboard}[具体例子：LQR问题]

\textbf{线性动力学}：$\dot{\mathbf{x}} = A\mathbf{x} + B\mathbf{u}$

\textbf{二次代价}：
\[
J = \frac{1}{2}\mathbf{x}(T)^T Q_f \mathbf{x}(T) + \frac{1}{2}\int_0^T (\mathbf{x}^T Q \mathbf{x} + \mathbf{u}^T R \mathbf{u}) \, dt
\]

\textbf{数值例子}：一维情况

$\dot{x} = x + u$（不稳定系统）

$J = \frac{1}{2}x(T)^2 + \frac{1}{2}\int_0^T (x^2 + u^2) \, dt$

\textbf{目标}：控制 $x$ 到 0，同时不用太大的 $u$

\end{blackboard}

\begin{blackboard}[Track A · Step 1：增广作用量与 Hamiltonian]

E-L 节我们把"$v_k = (q_{k+1}-q_k)/\Delta t$"用乘子 $\lambda_k$ 绑定。
现在把"$\dot x = f(x, u, t)$"用乘子 $\lambda(t)$ 绑定——\textbf{连续时间版本}。

\textbf{增广作用量}：
\[
\bar J[x, u, \lambda] = \phi(x(T)) + \int_0^T \Big[L(x, u, t) + \lambda(t)^\top\!\big(f(x, u, t) - \dot x\big)\Big] dt
\]

\textbf{定义 Hamiltonian}：
\[
\boxed{\;H(x, u, \lambda, t) := L(x, u, t) + \lambda^\top f(x, u, t)\;}
\]

\textbf{分部积分}把 $\dot x$ 上的导数挪到 $\lambda$ 上（保留端点项）：
\[
\bar J = \phi(x(T)) - \lambda(T)^\top x(T) + \lambda(0)^\top x_0 + \int_0^T \big[H(x, u, \lambda, t) + \dot\lambda^\top x\big]\, dt
\]
自由变量：$x(t), u(t), \lambda(t)$ 加上终端 $x(T)$（终端状态不固定）；$x(0) = x_0$ 固定。

\end{blackboard}

\begin{blackboard}[Track A · Step 2：变分 $\to$ Pontryagin 最大值原理（PMP）]

对每组变量求变分、设为零：

\textbf{对 $x(t)$ 的变分}（$0 < t < T$）：
\[
\frac{\partial H}{\partial x} + \dot\lambda = 0
\;\;\Longrightarrow\;\;
\boxed{\,\dot\lambda = -\frac{\partial H}{\partial x}\,}
\quad\text{（costate 方程，反向时间）}
\]

\textbf{对 $u(t)$ 的变分}：
\[
\frac{\partial H}{\partial u} = 0
\;\;\Longrightarrow\;\;
\boxed{\,u^*(t) = \arg\min_u\, H(x(t), u, \lambda(t), t)\,}
\]
（一般 PMP 是"最小值原理"——$u^*$ 让 $H$ 取最小，即便控制有界等不可导情形也成立。）

\textbf{对 $\lambda(t)$ 的变分}：恢复动力学 $\dot x = f(x, u, t)$。

\textbf{对 $x(T)$ 的变分}（终端自由）：
\[
\nabla_x\phi\big|_T - \lambda(T) = 0
\;\;\Longrightarrow\;\;
\boxed{\,\lambda(T) = \nabla_x \phi(x(T))\,}\quad\text{（横截性条件）}
\]

\textbf{合起来 —— PMP}：在最优轨迹上存在 $\lambda(t)$ 使
\[
\dot x = \frac{\partial H}{\partial \lambda},\quad
\dot\lambda = -\frac{\partial H}{\partial x},\quad
u^* = \arg\min_u H,\quad
\lambda(T) = \nabla_x\phi.
\]
这是 \textbf{Hamilton 方程组 $+$ 控制最小化}——E-L 在带控制问题上的精确推广。

\end{blackboard}

\begin{blackboard}[Track B · Step 1：Bellman 最优性原理]

\textbf{关键洞察}（Bellman, 1957）：

``最优策略的子策略也是最优的''

\textbf{数学表述}：定义值函数

\[
V(\mathbf{x}, t) = \min_{\mathbf{u}(\cdot)} \left[ \phi(\mathbf{x}(T)) + \int_t^T L(\mathbf{x}, \mathbf{u}, s) \, ds \right]
\]

从状态 $(\mathbf{x}, t)$ 出发的最小代价

\textbf{最优性原理}：
\[
V(\mathbf{x}, t) = \min_{\mathbf{u}} \left[ L(\mathbf{x}, \mathbf{u}, t) \, dt + V(\mathbf{x} + f(\mathbf{x}, \mathbf{u}, t) \, dt, t + dt) \right]
\]

\end{blackboard}

\begin{blackboard}[Track B · Step 2：Taylor 展开 $\to$ HJB PDE]

\textbf{Taylor展开}：
\[
V(\mathbf{x} + f \, dt, t + dt) \approx V + \frac{\partial V}{\partial t} dt + \nabla_x V \cdot f \, dt
\]

\textbf{代入最优性原理}：
\[
V = \min_{\mathbf{u}} \left[ L \, dt + V + \frac{\partial V}{\partial t} dt + \nabla_x V \cdot f \, dt \right]
\]

\textbf{消去 $V$，除以 $dt$}：
\[
0 = \min_{\mathbf{u}} \left[ L + \frac{\partial V}{\partial t} + \nabla_x V \cdot f \right]
\]

\end{blackboard}

\begin{blackboard}[Hamilton-Jacobi-Bellman方程]

\begin{keypoint}[HJB方程]
\[
\boxed{-\frac{\partial V}{\partial t} = \min_{\mathbf{u}} \left[ L(\mathbf{x}, \mathbf{u}, t) + \nabla_x V \cdot f(\mathbf{x}, \mathbf{u}, t) \right]}
\]
边界条件：$V(\mathbf{x}, T) = \phi(\mathbf{x})$
\end{keypoint}

\textbf{这是一个偏微分方程（PDE）}

\textbf{从终端 $t = T$ 向初始 $t = 0$ 求解}（\textbf{反向}！）

\textbf{难点}：维度灾难——$V$ 定义在状态空间上，高维时难以求解

\end{blackboard}

\begin{blackboard}[桥接：$\lambda(t) = \nabla_x V(x^*(t), t)$]

\textbf{断言}：PMP 的 costate $\lambda(t)$ 恰好等于 HJB 的值函数梯度 $\nabla_x V$ 在最优轨迹上的取值。

\textbf{推导（沿最优轨迹）}：把 $V(x^*(t), t)$ 对时间微分
\[
\frac{dV}{dt} = \frac{\partial V}{\partial t} + \nabla_x V \cdot \dot x^*
\]
HJB 方程在最优 $u^*$ 处取等（$\min$ 实现）：
\[
\frac{\partial V}{\partial t} + L(x^*, u^*, t) + \nabla_x V \cdot f(x^*, u^*, t) = 0
\]
两式相减：
\[
\frac{dV}{dt} = -L(x^*, u^*, t)
\]
即 $V$ 沿最优轨迹\textbf{以瞬时代价 $L$ 在被消耗}——这正是值函数的定义。

\textbf{识别 $\lambda$}：HJB 写成 $L + V_t + \nabla_x V\cdot f = 0$，对比 PMP 的 Hamiltonian
$H = L + \lambda^\top f$ 和 costate 方程 $\dot\lambda = -\partial H/\partial x$，
可以证明（沿任意小扰动）
\[
\boxed{\;\lambda(t) = \nabla_x V(x^*(t), t)\;}
\]
更精确地：
\begin{itemize}
    \item PMP 的 costate ODE $\dot\lambda = -\partial H/\partial x$ \textbf{就是 HJB 的特征线方程}
    \item HJB 沿最优轨迹"投影"成 PMP——一条 PDE 在一条曲线上塌缩成 ODE
    \item 终端 $\lambda(T) = \nabla_x\phi$ 和 $V(\cdot, T) = \phi$ \textbf{的梯度} 完全一致
\end{itemize}

\textbf{所以两条路得到的是同一个 $\lambda$}——继承自 E-L 节运动学约束的乘子。

\end{blackboard}

\begin{keypoint}[PMP vs HJB：维度对偶]

\renewcommand{\arraystretch}{1.2}
\begin{tabular}{p{0.22\linewidth}|p{0.34\linewidth}|p{0.34\linewidth}}
\toprule
 & \textbf{PMP（Track A）} & \textbf{HJB（Track B）} \\
\midrule
对象 & 一条最优轨迹 $(x^*, \lambda^*)$ & 整个值场 $V(x, t)$ \\
\midrule
类型 & ODE 系统于 $\mathbb{R}^{2n}$ & PDE 于 $\mathbb{R}^{n+1}$ \\
\midrule
求解方向 & 两点边值（$x(0)$ 前、$\lambda(T)$ 后） & 反向时间（终值 $V(\cdot,T)=\phi$） \\
\midrule
维度灾难 & 无——只算单条轨迹 & 有——$V$ 在全空间存储 \\
\midrule
在 RL 里 & model-based / DDP / iLQR / 神经 ODE & value-based / Bellman / TD / Q-learning \\
\midrule
$\lambda$ 的化身 & 乘子（继承自 E-L 节） & $\nabla_x V$（值函数梯度） \\
\bottomrule
\end{tabular}

\vspace{0.3em}
\textbf{这俩是 Legendre 变换的一对}——PMP 是 Lagrangian 视角（$x, \dot x$ 配对），
HJB 是 Hamiltonian 视角（$x, p$ 配对，$p = \nabla_x V$）。
\textbf{和单摆那节铺的 Lagrange–Hamilton 二象性是同一个东西，搬到了带控制的情形}。

\end{keypoint}

\begin{blackboard}[LQR的HJB解]

\textbf{猜测}：$V(\mathbf{x}, t) = \frac{1}{2}\mathbf{x}^T P(t) \mathbf{x}$

\textbf{代入HJB}：

$\nabla_x V = P\mathbf{x}$

$\frac{\partial V}{\partial t} = \frac{1}{2}\mathbf{x}^T \dot{P} \mathbf{x}$

\textbf{最小化对 $\mathbf{u}$}：

$\mathbf{u}^* = -R^{-1} B^T P \mathbf{x}$（线性反馈！）

\textbf{代入得 Riccati 方程}：
\[
-\dot{P} = Q + A^T P + PA - PBR^{-1}B^T P
\]

边界条件：$P(T) = Q_f$

\end{blackboard}

\begin{blackboard}[Track A 应用：LQR via PMP（得到同一个 Riccati）]

把 PMP 套到 LQR。Hamiltonian：
\[
H = \tfrac{1}{2}(x^\top Q x + u^\top R u) + \lambda^\top(Ax + Bu)
\]

\textbf{PMP 三条件}：
\begin{align*}
\dot\lambda &= -\partial_x H = -Qx - A^\top \lambda \\
\partial_u H &= R u + B^\top \lambda = 0
\quad\Longrightarrow\quad u^* = -R^{-1} B^\top \lambda \\
\dot x &= Ax + Bu = Ax - B R^{-1} B^\top \lambda
\end{align*}
横截性：$\lambda(T) = Q_f\, x(T)$。

\textbf{Ansatz}：$\lambda(t) = P(t)\, x(t)$（类比 HJB 那一支的 $V = \tfrac{1}{2}x^\top P x$）。

代入 $\dot\lambda = \dot P\, x + P\,\dot x$，并结合上面的 ODE：
\[
\dot P\, x + P\big(Ax - B R^{-1} B^\top P x\big) = -Q x - A^\top P x
\]
$x$ 任意，约去得到\textbf{同一个 Riccati 方程}：
\[
-\dot P = Q + A^\top P + P A - P B R^{-1} B^\top P
\]
边界 $P(T) = Q_f$ ——和 Track B 完全一致。

\textbf{$\lambda = \nabla_x V$ 在 LQR 上的具象}：
\[
\underbrace{\lambda = P x}_{\text{PMP 的 ansatz}}
\;=\;
\underbrace{\nabla_x V = \nabla_x \big(\tfrac{1}{2} x^\top P x\big) = P x}_{\text{HJB 那条线}}
\]
\textbf{同一个 $P$，同一个 $\lambda$}——两条路真的汇合了。

\end{blackboard}

\begin{blackboard}[LQR数值例子]

\textbf{一维}：$\dot{x} = x + u$，$Q = R = Q_f = 1$

\textbf{Riccati方程}：$-\dot{P} = 1 + 2P - P^2$

\textbf{稳态}（$\dot{P} = 0$）：$P^2 - 2P - 1 = 0$

$P = 1 + \sqrt{2} \approx 2.414$

\textbf{最优控制}：$u^* = -Px = -2.414x$

\textbf{闭环动力学}：$\dot{x} = x - 2.414x = -1.414x$

\textbf{稳定！}（极点在 $-1.414$）

\textbf{不加控制}：$\dot{x} = x$，指数增长，不稳定

\end{blackboard}

\begin{keypoint}[核心公式]

\textbf{1. Euler-Lagrange方程}：
\[
\frac{\partial F}{\partial y} - \frac{d}{dx}\frac{\partial F}{\partial y'} = 0
\]

\textbf{2. 最大熵分布}：
\[
p(x) \propto \exp\left( -\sum_k \lambda_k f_k(x) \right)
\]

\textbf{3. ELBO}：
\[
\text{ELBO} = \mathbb{E}_q[\log p(\mathbf{x}|\mathbf{z})] - D_{\text{KL}}(q \| p)
\]

\textbf{4. HJB方程}：
\[
-\frac{\partial V}{\partial t} = \min_{\mathbf{u}} \left[ L + \nabla V \cdot f \right]
\]

\textbf{5. Pontryagin 最大值原理（PMP）}：
\[
\dot\lambda = -\partial_x H,\quad u^* = \arg\min_u H,\quad H = L + \lambda^\top f
\]
其中 $\lambda(t) = \nabla_x V(x^*(t), t)$ 把 PMP 和 HJB 桥接。

\end{keypoint}

\begin{keypoint}[核心联系]

\begin{center}
\begin{tabular}{c|c}
概念 & 联系 \\
\hline
E-L方程 & 泛函优化的必要条件（KKT 的连续极限） \\
最大熵 & $\to$ 指数族分布 \\
变分推断 & min KL $\Leftrightarrow$ max ELBO \\
HJB方程 & $=$ PMP 的"值场"对偶；$\lambda = \nabla_x V$
\end{tabular}
\end{center}

\textbf{统一视角}：拉格朗日乘子 $\lambda$ 贯穿全章——力学的共轭动量、控制的 costate、HJB 的 value 梯度，是同一个对象的不同语言。

\end{keypoint}
